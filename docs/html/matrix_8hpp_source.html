1018 template <typename E1, typename E2> auto operator|(E1 const &u, E2 const &v) {
1019  if (u.get_dimension().col_dimen != v.get_dimension().row_dimen) {
1020  throw std::logic_error(
1021  std::string(
1022  "Dot product cannot be called on matrices with dimension ") +

1024  v.get_dimension().to_string());
1025  }
1026  matrix<decltype(u.get(0, 0))> ans(u.get_dimension().row_dimen,
1027  v.get_dimension().col_dimen);
1028  for (size_t i = 0; i < u.get_dimension().row_dimen; i++) {
1029  for (size_t j = 0; j < v.get_dimension().col_dimen; j++) {
1030  ans.get(i, j) = 0;
1031  for (size_t k = 0; k < v.get_dimension().row_dimen; k++)
1032  ans.get(i, j) += u.get(i, k) * v.get(k, j);
1033  }
1034  }
1035  return ans;
